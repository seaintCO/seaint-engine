app/api/apify/start/route.ts 
import { NextResponse } from 'next/server';

export async function POST(request: Request) {
  try {
    const body = await request.json();
    const { searchQuery, location, maxResults } = body;
    
    const APIFY_TOKEN = process.env.APIFY_TOKEN;
    const ACTOR_ID = process.env.APIFY_ACTOR_ID;

    if (!APIFY_TOKEN || !ACTOR_ID) {
      return NextResponse.json({ error: 'Missing Apify Credentials' }, { status: 500 });
    }

    const response = await fetch(
      `https://api.apify.com/v2/acts/${ACTOR_ID}/runs?token=${APIFY_TOKEN}`,
      {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ search: searchQuery, location, maxResults: parseInt(maxResults) || 50 }),
      }
    );

    const data = await response.json();
    if (!response.ok) throw new Error(data.error?.message || 'Failed to start run');

    return NextResponse.json({ 
      runId: data.data.id, 
      status: data.data.status 
    });

  } catch (error: any) {
    return NextResponse.json({ error: error.message }, { status: 500 });
  }
}
app/api/apify/status/route.ts 
import { NextResponse } from 'next/server';

export async function GET(request: Request) {
  const { searchParams } = new URL(request.url);
  const runId = searchParams.get('runId');
  const APIFY_TOKEN = process.env.APIFY_TOKEN;

  if (!runId || !APIFY_TOKEN) {
    return NextResponse.json({ error: 'Missing runId or token' }, { status: 400 });
  }

  try {
    const response = await fetch(
      `https://api.apify.com/v2/actor-runs/${runId}?token=${APIFY_TOKEN}`,
      { cache: 'no-store' }
    );
    const data = await response.json();
    
    return NextResponse.json({ 
      status: data.data.status, 
      defaultDatasetId: data.data.defaultDatasetId 
    });
  } catch (error: any) {
    return NextResponse.json({ error: error.message }, { status: 500 });
  }
}
app/api/apify/results/route.ts 
import { NextResponse } from 'next/server';

export async function GET(request: Request) {
  const { searchParams } = new URL(request.url);
  const datasetId = searchParams.get('datasetId');
  const limit = searchParams.get('limit') || '100';
  const offset = searchParams.get('offset') || '0';
  
  const APIFY_TOKEN = process.env.APIFY_TOKEN;

  if (!datasetId || !APIFY_TOKEN) {
    return NextResponse.json({ error: 'Missing datasetId or token' }, { status: 400 });
  }

  try {
    const response = await fetch(
      `https://api.apify.com/v2/datasets/${datasetId}/items?token=${APIFY_TOKEN}&format=json&clean=true&limit=${limit}&offset=${offset}`,
      { cache: 'no-store' }
    );
    const items = await response.json();
    
    return NextResponse.json({ items });
  } catch (error: any) {
    return NextResponse.json({ error: error.message }, { status: 500 });
  }
}

Vercel Deployment Guide

  1. Push your Next.js project to GitHub.
  2. Import the repository into Vercel.
  3. In the project settings, navigate to Environment Variables.
  4. Add APIFY_TOKEN and paste your personal API token.
  5. Add APIFY_ACTOR_ID and paste the specific Google Maps/Scraper Actor ID.
  6. Deploy the project. The API routes above securely execute on edge/serverless functions.
